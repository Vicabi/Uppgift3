                        private boolean isSolvable() {
                          int countInversions = 0;

                          for (int i = 0; i < nbTiles; i++) {
                            for (int j = 0; j < i; j++) {
                              if (tiles[j] > tiles[i])
                                countInversions++;
                            }
                          }

                          return countInversions % 2 == 0;
